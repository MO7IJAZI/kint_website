function forceSpaceWrapping(html) {
  if (!html) return "";
  
  // Replace all spaces with non-breaking spaces inside specific tags
  const processed = html
    // Wrap all text nodes in span with specific class
    .replace(/>([^<]+)</g, (match, text) => {
      if (text.trim() === '') return match;
      return `><span class="force-space-wrap">${text}</span><`;
    })
    // Add zero-width space after punctuation
    .replace(/([.,:;!?])([^<\s])/g, '$1&#8203;$2')
    // Ensure URLs don't break layout
    .replace(/(https?:\/\/[^\s<]+)/g, '<span class="force-url-break">$1</span>');
  
  return processed;
}

## **3. Key Changes in This Approach:**

1. **Removed all Quill editor classes** - They might be conflicting
2. **Used simple `.article-content` container** - No complex class inheritance
3. **Force inherited styles on ALL children** - Using `!important` and `inherit`
4. **Added `text-align: justify` and `text-justify: inter-word`** - Forces space-based wrapping
5. **Used `white-space: normal`** - Ensures normal whitespace handling
6. **Added the `forceSpaceWrapping` utility** - Preprocesses content to add zero-width spaces after punctuation

## **How It Works:**

1. When text hits the container boundary
2. The browser looks for the **last space** in the line
3. It breaks the line at that space
4. Continues the remaining text on the next line
5. Punctuation stays with its word
